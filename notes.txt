  describe 'minutes#create action' do
    it 'should add the minutes to the database' do 
      user = FactoryBot.create(:user)
      sign_in user

      post :create, params: { minute: {
        title: 'Test',
        date: '2018-09-30',
        pdf_file: 'pdf-file'
      } }

      minute = Minute.last
      expect(response).to be_ok
      expect(Minute.count).to eq 1
      expect(minute.title).to eq('Test')
      expect(minute.user).to eq(user)
      expect(minute.date).to eq('2018-09-30')
    end
  end


  describe 'minutes#edit action' do
    it 'should redirect non-admin from the edit page' do
      user = FactoryBot.create(:user)
      minute = FactoryBot.create(:minute)
      sign_in user

      get :edit, params: { id: minute.id }
      expect(response).to redirect_to root_path
    end
  end
